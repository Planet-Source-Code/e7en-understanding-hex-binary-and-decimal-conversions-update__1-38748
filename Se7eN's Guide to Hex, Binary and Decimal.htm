Today we will look at the ASCII (‘American Standard Code for Information Interchange’) coding scheme.

 

 

As most people know computers work with 2 digits: 0 and 1. These digits are called bits.

ASCII uses 7 bits to represent a character with a leading 0 to create an 8-bit code to make data transfer more compatible with today’s technology.

Converting in VB

 

Great now you know how it works you want to know how the hell to do it in VB.

Well here are some Functions that will easily convert them for you.

 

‘====================Hex To ASCII====================’

Function HexToAscii(str As String) As String

Dim TempStr As String

Dim X As Integer

 

If (Len(str)) Mod 2 <> 0 Then

MsgBox "Invalid Hex Values"

Exit Function

End If

 

For X = 1 To Len(str) Step 2

TempStr = TempStr & Chr(Val("&h" & Mid(str, X, 2)))

Next

 

HexToAscii = TempStr

End Function

 

 

‘====================ASCII To Hex====================’

Function AsciiToHex(str As String) As String

Dim TempStr As String

Dim X As Integer

 

For X = 1 To Len(str)

TempStr = TempStr & Hex(Asc(Mid(str, X, 1)))

Next

 

AsciiToHex = TempStr

End Function

 

‘====================ASCII To Dec====================’

Function AsciiToDecimal(str As String) As String

Dim TempStr As String

Dim X As Integer

 

For X = 1 To Len(str)

TempStr = TempStr & Asc(Mid(str, X, 1))

Next

 

AsciiToDecimal = TempStr

End Function

 

‘=================Hex To Binary=====================’

Function HexToBinary(str As String, Optional Seperator As String = "")

Dim TempStr As String

Dim X As Integer

 

If (Len(str)) Mod 2 <> 0 Then

MsgBox "Invalid Hex Values"

Exit Function

End If

 

For X = 1 To Len(str)

    Select Case Mid(str, X, 1)

        Case "0"

            TempStr = TempStr & "0000"

        Case "1"

            TempStr = TempStr & "0001"

        Case "2"

            TempStr = TempStr & "0010"

        Case "3"

            TempStr = TempStr & "0011"

        Case "4"

            TempStr = TempStr & "0100"

        Case "5"

            TempStr = TempStr & "0101"

        Case "6"

            TempStr = TempStr & "0110"

        Case "7"

            TempStr = TempStr & "0111"

        Case "8"

            TempStr = TempStr & "1000"

        Case "9"

            TempStr = TempStr & "1001"

        Case "A"

            TempStr = TempStr & "1010"

        Case "B"

            TempStr = TempStr & "1011"

        Case "C"

            TempStr = TempStr & "1100"

        Case "D"

            TempStr = TempStr & "1101"

        Case "E"

            TempStr = TempStr & "1110"

        Case "F"

            TempStr = TempStr & "1111"

    End Select

    TempStr = TempStr & Seperator

Next

 

HexToBinary = TempStr

End Function

 

‘=================Binary To Hex=====================’

Function BinaryToHex(str As String) As String

Dim TempStr As String

Dim X As Integer

 

If (Len(str)) Mod 8 <> 0 Then

    End Select

   

Next

 

BinaryToHex = TempStr

End Function

 

‘=================Remove Seperators==================’

Function RemoveSeperator(str As String) As String

Dim X As Integer

Dim TempStr As String

 

For X = 1 To Len(str)

    If Mid(str, X, 1) = 1 Or Mid(str, X, 1) = 0 Then TempStr = TempStr & Mid(str, X, 1)

Next

 

RemoveSeperator = TempStr

End Function
